void bSort(int a[], int size);

	unsigned int j  = 0;    //loop control

	puts("Enter 20 integers between 10 and 100");

	for(i = 0; i < SIZE; i++)
	    scanf("%d", &nums[i]);

                  duplicate = 1;

              if( i == 0)
                 duplicate = 0;

             if(duplicate == 0 )
                 nums2[i] = nums[i];

	       duplicate = 0; //reset to 0;
	} //end for loop

	puts("");//a new line

	for(i = 0; i < SIZE; i++){
		if( nums2[i] != 0)
		printf("%d ", nums2[i]);

	puts(""); //a new line

	//addone function that sorts the arrays from smallest to largest
	        //and then display that

	return 0;
}

void bSort(int a[], int size){
	
	int pass = 0;
	int k = 0;
	int hold = 0;
	
	for( pass = 1; pass < size; pass++){
		
		for ( k = 0; k < size -1; k++){
			if( a[k] > a[k + 1] ){
				hold = a[k];
				a[k] = a[k+1];
				a[k+1] = hold;
			}//end if block
		}//end for loop
	}//end for loop
	
}//end function bSort
